/* Starting time:
*
* 10/23/2017 @ 12:00 a.m. US Eastern
*/

/* Cycle:
*
* 12 days
* 288 hours
* 17280 minutes
*/

// TODO: consider changing starting point to UTC and localizing from that

starting_point = localize(new Date(US Eastern 10/23/2017 @ 12:00 a.m.))
current_time = now()
time_passed = current_time - starting_point
minutes_passed = to_minutes(time_passed)

current_minute = minutes_passed % 150
if current_minute <= 100 then
  phase = day
else
  phase = night

if phase == day then
  remaining_minutes = 100 - current_minute
else
  remaining_minutes = 50 - current_minute

display It is currently {phase}
display {remaining_minutes} until {night if phase == day else day}
